import tkinter as tk from tkinter import ttk, messagebox import webbrowser import os # ------------------- Funções das Páginas ------------------- def abrir_pagina_simples(titulo, texto): janela = tk.Toplevel() janela.title(titulo) janela.geometry("500x400+500+200") frame = tk.Frame(janela) frame.pack(expand=True) label = tk.Label(frame, text=texto, font=("Arial", 14)) label.pack(pady=20) btn_voltar = tk.Button(frame, text="Voltar", width=15, command=janela.destroy) btn_voltar.pack(pady=20) def abrir_acesso_remoto(): janela = tk.Toplevel() janela.title("Acesso Remoto") janela.geometry("600x400+500+200") frame = tk.Frame(janela) frame.pack(expand=True) titulo = tk.Label(frame, text="Acesso Remoto", font=("Arial", 18, "bold")) titulo.pack(pady=10) label_vnc = tk.Label(frame, text="Clique abaixo para baixar o VNC Viewer:", font=("Arial", 12)) label_vnc.pack(pady=10) def abrir_vnc(): webbrowser.open("https://www.realvnc.com/en/connect/download/viewer/?lai_vid=l2zKNkO3qHy8l&lai_sr=5-9&lai_sl=l") btn_vnc = tk.Button(frame, text="Baixar VNC Viewer", width=25, command=abrir_vnc) btn_vnc.pack(pady=10) btn_voltar = tk.Button(frame, text="Voltar", width=15, command=janela.destroy) btn_voltar.pack(pady=20) def abrir_lojas(): abrir_pagina_simples("Lojas", "Página de Lojas em construção...") def abrir_contatos(): abrir_pagina_simples("Contatos", "Contatos da equipe e setores...") def abrir_links(): abrir_pagina_simples("Links de Acesso", "Links úteis e principais sistemas...") def abrir_menuboard(): abrir_pagina_simples("MenuBoard", "Configurações e acesso ao MenuBoard...") def abrir_aplicativos(): abrir_pagina_simples("Aplicativos", "Aplicativos disponíveis para instalação...") # ------------------- JANELA PRINCIPAL ------------------- root = tk.Tk() root.title("Painel Principal KFC") root.geometry("600x500+450+150") root.resizable(False, False) frame_principal = tk.Frame(root) frame_principal.pack(expand=True) label_titulo = tk.Label(frame_principal, text="Painel de Atalhos", font=("Arial", 22, "bold")) label_titulo.pack(pady=20) # ------------------- Botões ------------------- botoes = [ ("Aplicativos", abrir_aplicativos), ("Contatos", abrir_contatos), ("Links de Acesso", abrir_links), ("Lojas", abrir_lojas), ("MenuBoard", abrir_menuboard), ("Acesso Remoto", abrir_acesso_remoto), ] for texto, comando in botoes: btn = tk.Button(frame_principal, text=texto, width=20, height=2, command=comando) btn.pack(pady=5) # ------------------- Loop Tkinter ------------------- root.mainloop()
